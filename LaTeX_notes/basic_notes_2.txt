depth
the value has to be set in the preamble and applies to the whole document
\setcounter{tocdepth}{0} - shows nothing
\setcounter{tocdepth}{1} - + section
\setcounter{tocdepth}{2} - + subsection
\setcounter{tocdepth}{3} - + subsubsection
\setcounter{tocdepth}{4} - + paragraph
\setcounter{tocdepth}{5} - + subparagraph

if you don't want to change depth for all sections, set tocdepth  for each section from which point you want to change
\addtocontents{toc}{\setcounter{todepth}{1}}
\section - included
\subsection - not included
\subsubsection
...
\addtocontents{toc}{\setcounter{todepth}{3}}
from here on till changed again, depth of sections is 3

    language=C,                % Choose the language of the code
    basicstyle=\ttfamily,      % Use a monospace font
    numbers=left,              % Line numbers on the left
    numberstyle=\tiny,         % Line numbers in tiny font
    stepnumber=1,              % Line numbers every 1 line
    frame=single,              % Add a frame around the code
    captionpos=b,              % Position the caption below the code
    breaklines=true,           % Enable line breaking
    tabsize=4                  % Set tab size
}
